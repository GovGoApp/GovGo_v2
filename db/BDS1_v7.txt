-- Views
CREATE VIEW public.vw_contratos_por_fornecedor AS
 SELECT ni_fornecedor,
    nome_razao_social_fornecedor,
    count(numero_contrato_empenho) AS contrato_count
   FROM contrato ct
  GROUP BY ni_fornecedor, nome_razao_social_fornecedor;;
CREATE VIEW public.vw_fornecedores AS
 SELECT ni_fornecedor AS cnpj,
    nome_razao_social_fornecedor AS razao_social,
    count(*) AS contrato_count
   FROM contrato ct
  WHERE ni_fornecedor ~ '^[0-9]{14}$'::text
  GROUP BY ni_fornecedor, nome_razao_social_fornecedor
 HAVING count(*) > 3;;
CREATE VIEW public.vw_fornecedores_pendentes AS
 WITH base AS (
         SELECT ct.ni_fornecedor AS cnpj,
            ct.nome_razao_social_fornecedor AS razao_social,
            count(*) AS contrato_count
           FROM contrato ct
          WHERE ct.ni_fornecedor ~ '^[0-9]{14}$'::text
          GROUP BY ct.ni_fornecedor, ct.nome_razao_social_fornecedor
         HAVING count(*) > 3
        )
 SELECT b.cnpj,
    b.razao_social,
    b.contrato_count
   FROM base b
     LEFT JOIN empresa e ON e.cnpj = b.cnpj
  WHERE e.cnpj IS NULL OR e.api_json IS NULL
  ORDER BY b.contrato_count DESC;;
CREATE VIEW public.vw_index_build_progress AS
 SELECT s.pid,
    ( SELECT pg_class.relname
           FROM pg_class
          WHERE pg_class.oid = s.relid) AS table_name,
        CASE
            WHEN s.index_relid = 0::oid THEN '-'::name
            ELSE ( SELECT pg_class.relname
               FROM pg_class
              WHERE pg_class.oid = s.index_relid)
        END AS index_name,
    s.phase,
    s.tuples_done,
    s.tuples_total,
        CASE
            WHEN s.tuples_total > 0 THEN round(100.0 * s.tuples_done::numeric / s.tuples_total::numeric, 2)
            ELSE NULL::numeric
        END::numeric(7,2) AS pct_tuples,
    s.blocks_done,
    s.blocks_total,
        CASE
            WHEN s.blocks_total > 0 THEN round(100.0 * s.blocks_done::numeric / s.blocks_total::numeric, 2)
            ELSE NULL::numeric
        END::numeric(7,2) AS pct_blocks,
    sa.state,
    sa.wait_event_type,
    sa.wait_event,
    sa.query_start
   FROM pg_stat_progress_create_index s
     LEFT JOIN pg_stat_activity sa ON sa.pid = s.pid
  WHERE s.relid = 'contrato_emb'::regclass::oid;;
